import cv2
import numpy as np

# Carregar imagem
img = cv2.imread('sua_foto.jpg')

# Converter para HSV (melhor para trabalhar com cores)
hsv = cv2.cvtColor(img, cv2.COLOR_BGR2HSV)

# Definir ranges para vermelho (o vermelho tem 2 ranges no HSV)
# Range 1: vermelhos mais claros
lower_red1 = np.array([0, 100, 100])
upper_red1 = np.array([10, 255, 255])

# Range 2: vermelhos mais escuros
lower_red2 = np.array([160, 100, 100])
upper_red2 = np.array([180, 255, 255])

# Criar máscaras
mask1 = cv2.inRange(hsv, lower_red1, upper_red1)
mask2 = cv2.inRange(hsv, lower_red2, upper_red2)

# Combinar máscaras
mask_red = cv2.bitwise_or(mask1, mask2)

# Aplicar máscara na imagem original
result = cv2.bitwise_and(img, img, mask=mask_red)

cv2.imshow('Original', img)
cv2.imshow('Apenas Vermelhos', result)
cv2.waitKey(0)
cv2.destroyAllWindows()
